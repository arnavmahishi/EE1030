\begin{align}
\text{General eqn: }n^Tx=0\\
n=\myvec{m\\-1}\implies n=\myvec{0\\-1}\\
\implies\myvec{0&-1}\myvec{x-\myvec{2\\-4}}=0\\
\implies\myvec{0&1}x=\myvec{0&1}\myvec{2\\-4}\\
\therefore\text{ The desired eqn: }\myvec{0&1}x=-4
\end{align}
